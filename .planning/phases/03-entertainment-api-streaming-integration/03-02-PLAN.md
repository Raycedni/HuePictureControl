            self._run_event = asyncio.Event()
            self._task: asyncio.Task | None = None
            self._state = "idle"

        @property
        def state(self) -> str:
            return self._state

        async def start(self, config_id: str) -> None:
            if self._state not in ("idle", "error"):
                return
